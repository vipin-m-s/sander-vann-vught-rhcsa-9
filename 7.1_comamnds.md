# tr  - translate
```
root@localhost:~# echo hello | tr a-z A-Z
HELLO
root@localhost:~# echo hello | tr [:lower:] [:upper:]
HELLO
```

# Cut 
- -d is delimiter
- -f for filed
```
root@localhost:~# cut -d : -f 1 /etc/passwd
root
bin
daemon
adm
lp
sync
shutdown
halt
mail
operator
games
ftp
nobody
dbus
yggdrasil
```

# sort
```
root@localhost:~# cut -d : -f 1 /etc/passwd | sort | head
adm
anna
```
